score = int(input('점수를 입력하세요 : '))

if score>100 :
    print('점수를 다시 입력해주세요')
else :
    if score >= 90 :
        print('A')
    elif score >= 80 :
        print('B')
    elif score >= 70 :
        print('C')
    else :
        print ('F')
-------------------------------------------------------------------------

int() = 자료형을 정수로 변형
str() = 자료형을 문자로 변형
float() = 자료형을 소수로 변형
